% Internal helper: image+caption in a fixed box
\newcommand{\galleryimg}[4]{%
  \begin{minipage}[t]{#1}
    \centering
    \if\relax\detokenize{#2}\relax
      % no height specified
      \includegraphics[width=\textwidth]{media/#3}%
    \else
      % height specified
      \includegraphics[width=\textwidth,height=#2,keepaspectratio=false]{media/#3}%
    \fi
    \if\relax\detokenize{#4}\relax
      % no caption
    \else
      \par\vspace{1ex}\small\textbf{#4}%
    \fi
  \end{minipage}%
}

\NewDocumentCommand{\fig}{ O{0.8\textwidth} O{} m O{} }{%
  \ifinfigs
    \galleryimg{#1}{#2}{#3}{#4}%
    \hfill
  \else
    \begin{figure}[H]
      \centering
      \if\relax\detokenize{#2}\relax
        \includegraphics[width=#1]{media/#3}%
      \else
        \includegraphics[width=#1,height=#2,keepaspectratio=false]{media/#3}%
      \fi
      \if\relax\detokenize{#4}\relax
        % no caption
      \else
        \caption*{\textbf{#4}}%
      \fi
    \end{figure}%
  \fi
}
